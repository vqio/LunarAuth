// Response codes
enum class AuthResponse {
    OK, APP_NOT_FOUND, APP_PAUSED, KEY_NOT_FOUND,
    KEY_PAUSED, KEY_EXPIRED, HWID_REQUIRED, HWID_MISMATCH, INVALID_SECRET
};

// Get HWID (gerado a partir do volume serial + MAC address)
std::string GetMachineHWID() {
    // Obtém o volume serial
    DWORD serial = 0;
    GetVolumeInformation("C:\\", NULL, 0, &serial, NULL, NULL, NULL, 0);
    
    // Obtém o endereço MAC
    std::string macAddress;
    IP_ADAPTER_INFO adapterInfo[16];
    DWORD dwBufLen = sizeof(adapterInfo);
    
    if (GetAdaptersInfo(adapterInfo, &dwBufLen) == ERROR_SUCCESS) {
        PIP_ADAPTER_INFO pAdapterInfo = adapterInfo;
        if (pAdapterInfo) {
            std::ostringstream oss;
            for (UINT i = 0; i < pAdapterInfo->AddressLength; i++) {
                oss << std::hex << std::setw(2) << std::setfill('0') 
                    << (int)pAdapterInfo->Address[i];
            }
            macAddress = oss.str();
        }
    }
    
    // Combina para criar um HWID único
    std::ostringstream hwid;
    hwid << std::hex << serial << "-" << macAddress;
    
    return hwid.str();
}

// Gera um hash SHA-256 (simplificado para exemplo)
std::string GenerateSignature(const std::string& secretId, const std::string& appId, 
                            const std::string& key, const std::string& hwid) {
    // Em produção, use uma biblioteca de hash adequada
    std::string data = secretId + appId + key + hwid;
    // Simulação simplificada - use uma biblioteca de hash real
    return "simulated_signature_" + data.substr(0, 10);
}

// Função para validar a chave
AuthResponse ValidateKey(const std::string& secretId,
                        const std::string& appId, 
                        const std::string& key, 
                        const std::string& hwid) {
    HINTERNET hSession = WinHttpOpen(L"LunarAuth C++ Client", 
                                   WINHTTP_ACCESS_TYPE_DEFAULT_PROXY,
                                   WINHTTP_NO_PROXY_NAME, 
                                   WINHTTP_NO_PROXY_BYPASS, 0);
    
    if (!hSession) return AuthResponse::KEY_NOT_FOUND;

    // Gera a assinatura de autenticação
    std::string signature = GenerateSignature(secretId, appId, key, hwid);

    // Conecta ao servidor (substitua pelo seu domínio)
    HINTERNET hConnect = WinHttpConnect(hSession, L"api.lunarauth.com", 
                                      INTERNET_DEFAULT_HTTPS_PORT, 0);
    if (!hConnect) {
        WinHttpCloseHandle(hSession);
        return AuthResponse::KEY_NOT_FOUND;
    }

    // Cabeçalhos da requisição
    LPCWSTR headers = L"Content-Type: application/json\r\n"
                     "X-Auth-Signature: " + signature.c_str() + L"\r\n"
                     "X-Auth-Secret: " + secretId.c_str() + L"\r\n"
                     "X-App-ID: " + appId.c_str() + L"\r\n";

    // Corpo da requisição
    std::wstring requestBody = L"{\"key\":\"" + 
                              std::wstring(key.begin(), key.end()) + 
                              L"\",\"hwid\":\"" + 
                              std::wstring(hwid.begin(), hwid.end()) + 
                              L"\"}";

    // Envia a requisição POST
    HINTERNET hRequest = WinHttpOpenRequest(hConnect, L"POST", 
                                          L"/api/v1/validate",
                                          NULL, WINHTTP_NO_REFERER, 
                                          WINHTTP_DEFAULT_ACCEPT_TYPES, 
                                          WINHTTP_FLAG_SECURE);

    if (!hRequest) {
        WinHttpCloseHandle(hConnect);
        WinHttpCloseHandle(hSession);
        return AuthResponse::KEY_NOT_FOUND;
    }

    // Envia os cabeçalhos
    if (!WinHttpSendRequest(hRequest, headers, -1L, 
                          (LPVOID)requestBody.c_str(), 
                          requestBody.size() * sizeof(wchar_t),
                          requestBody.size() * sizeof(wchar_t), 0)) {
        WinHttpCloseHandle(hRequest);
        WinHttpCloseHandle(hConnect);
        WinHttpCloseHandle(hSession);
        return AuthResponse::KEY_NOT_FOUND;
    }

    // Processa a resposta...
    // (código de processamento da resposta permanece o mesmo)
    // ...

    WinHttpCloseHandle(hRequest);
    WinHttpCloseHandle(hConnect);
    WinHttpCloseHandle(hSession);
    
    // Processa o status code
    if (statusCode == 200) return AuthResponse::OK;
    if (statusCode == 400) return AuthResponse::KEY_NOT_FOUND;
    if (statusCode == 401) return AuthResponse::INVALID_SECRET;
    if (statusCode == 403) return AuthResponse::HWID_MISMATCH;
    if (statusCode == 402) return AuthResponse::KEY_EXPIRED;
    
    return AuthResponse::KEY_NOT_FOUND;
}

int main() {
    // Configurações da sua conta
    const std::string SECRET_ID = "seu_secret_id_aqui"; // Substitua pelo seu Secret ID
    const std::string DEFAULT_APP_ID = "seu_app_id_aqui"; // AppID padrão
    
    std::string appId = DEFAULT_APP_ID;
    std::string key;
    std::string hwid = GetMachineHWID();
    
    std::cout << "=== LunarAuth C++ Client ===\n";
    std::cout << "HWID: " << hwid << "\n\n";
    
    std::cout << "Digite a chave: ";
    std::getline(std::cin, key);

    // Valida a chave
    AuthResponse result = ValidateKey(SECRET_ID, appId, key, hwid);

    // Processa o resultado
    switch (result) {
        case AuthResponse::OK:
            std::cout << "\n✅ Autenticação bem-sucedida!\n";
            // Código para quando a autenticação for bem-sucedida
            break;
            
        case AuthResponse::INVALID_SECRET:
            std::cout << "\n❌ Secret ID inválido.\n";
            break;
            
        case AuthResponse::KEY_NOT_FOUND:
            std::cout << "\n❌ Chave inválida ou não encontrada.\n";
            break;
            
        case AuthResponse::HWID_MISMATCH:
            std::cout << "\n❌ Esta chave já está em uso em outro dispositivo.\n";
            break;
            
        case AuthResponse::KEY_EXPIRED:
            std::cout << "\n❌ Esta chave expirou.\n";
            break;
            
        default:
            std::cout << "\n❌ Erro na autenticação.\n";
    }

    std::cout << "\nPressione Enter para sair...";
    std::cin.get();
    return 0;
}
